import tkinter as tk
import threading
import queue
import time
import math

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Queue Example")

        self.q = queue.Queue()
        self.stop_event = threading.Event()
        self.worker = None

        self.start_btn = tk.Button(root, text="Start", command=self.start)
        self.start_btn.pack(pady=5)

        self.stop_btn = tk.Button(root, text="Stop", command=self.stop)
        self.stop_btn.pack(pady=5)

        self.text = tk.Text(root, width=60, height=20)
        self.text.pack()

        self.root.after(100, self.check_queue)

    def producer(self):
        n = 1

        while not self.stop_event.is_set():

            # Some math work
            result = sum(math.sin(i * n) for i in range(10000))

            self.q.put(f"n={n} result={result:.4f}")

            n += 1

            time.sleep(0.2)

    def start(self):
        if self.worker and self.worker.is_alive():
            return

        self.stop_event.clear()

        self.worker = threading.Thread(
            target=self.producer,
            daemon=True
        )

        self.worker.start()

        self.text.insert(tk.END, "Started\n")

    def stop(self):
        self.stop_event.set()
        self.text.insert(tk.END, "Stopped\n")

    def check_queue(self):
        while not self.q.empty():
            msg = self.q.get()
            self.text.insert(tk.END, msg + "\n")
            self.text.see(tk.END)

        self.root.after(100, self.check_queue)

root = tk.Tk()
app = App(root)
root.mainloop()
